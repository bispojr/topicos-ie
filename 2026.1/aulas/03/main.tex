        \begin{frame}{Questão 001}
		\begin{block}{[Q001]}
			O Workshop de Ensino em Pensamento Computacional, Algoritmos e Programação (WAlgProg) foi um evento científico da Computação que existiu do ano de 2015 até 2019. Em qual das áreas da Computação o WAlgProg \underline{{\bf melhor}} se encaixa?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Redes de Computadores
			\item Educação em Computação
			\item Informática na Educação
			\item Banco de Dados
		\end{enumerate}
	\end{frame}

        \begin{frame}{Questão 002}
		\begin{block}{[Q002]}
			Podemos descrever como um dos objetivos da Educação em Computação:
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Criar ferramentas pedagógicas para o suporte de professores no ensino de uma forma geral.
			\item Construir sistemas de visualização de modelos atômicos utilizando a computação como mediador pedagógico.
			\item Aperfeiçoar os processos de ensino e aprendizagem da Computação como ciência.
			\item Desenvolver sistemas computacionais que possuam mecanismos de aprendizagem (semi-)automática.
		\end{enumerate}
	\end{frame}

        \begin{frame}{Questão 003}
		\begin{block}{[Q003]}
			São interesses da Educação em Computação, \underline{{\bf exceto}}:
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item uso de computação vestível na educação.
			\item mentoria profissional de graduandos de Computação.
			\item ensino da honestidade acadêmica na Computação.
			\item influência da língua materna em erros de compilação.
		\end{enumerate}
	\end{frame}

        \begin{frame}{Questão 004}
		\begin{block}{[Q004]}
			Um bom ponto de partida para a compreensão sobre o que seria Informática na Educação, a partir do que foi exposto por Valente (2008), seria:
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Aprendizagem de Computação
			\item Desenvolvimento através do Computador
			\item Ensino de Computação
			\item Ensino através do Computador
		\end{enumerate}
	\end{frame}

        \begin{frame}{Questão 005}
		\begin{block}{[Q005]}
			A definição de Informática na Educação apresentada por Bispo Jr. e colegas (2020) é:
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item o uso de artefatos tecnológicos como mediadores nos processos de ensino e aprendizagem, e aos seus impactos.
			\item o ensino de artefatos tecnológicos como mediadores nos processos de ensino e aprendizagem, e aos seus impactos.
			\item o uso de computadores como mediadores nos processos de ensino e aprendizagem, e aos seus impactos.
			\item o ensino de computadores como mediadores nos processos de ensino e aprendizagem, e aos seus impactos.
		\end{enumerate}
	\end{frame}

        \begin{frame}{Questão 006}
		\begin{block}{[Q006]}
			São tópicos de interesse da Informática na Educação, \underline{\bf exceto}:
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item Simuladores e Jogos Educativos
                \item Motivação de Graduandos na Computação
                \item  Aprendizagem  Colaborativa  Apoiada  por Computadores
                \item Mineração de Dados Educacionais
		\end{enumerate}
	\end{frame}
